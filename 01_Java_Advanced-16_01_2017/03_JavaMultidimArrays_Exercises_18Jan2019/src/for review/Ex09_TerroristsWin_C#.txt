using System;
 
public class TerroristWin
{
    public static void Main()
    {
        string input = Console.ReadLine();
 
        char[] inputStringAsCharArray = input.ToCharArray();
        int startIndex = input.IndexOf('|', 0);
        int endIndex = 0;
        int asciSum = 0;
        int bombPower = 0;
 
        while (startIndex != -1)
        {
            asciSum = 0;
            endIndex = input.IndexOf('|', startIndex + 1);
 
            if (endIndex <= startIndex)
            {
                break;
            }
 
            for (int i = startIndex + 1; i < endIndex; i++)
            {
                asciSum += inputStringAsCharArray[i];
                inputStringAsCharArray[i] = '.';
 
            }
 
            string asciSumAsString = asciSum.ToString();
            string bombPowerAsString = asciSumAsString[asciSumAsString.Length - 1].ToString();
            bombPower = int.Parse(bombPowerAsString);
 
            for (int i = startIndex; i >= startIndex - bombPower; i--)
            {
                if (i < 0)
                {
                    break;
                }
 
                inputStringAsCharArray[i] = '.';
            }
 
            for (int i = endIndex; i <= endIndex + bombPower; i++)
            {
                if (i > inputStringAsCharArray.Length - 1)
                {
                    break;
                }
 
                inputStringAsCharArray[i] = '.';
            }
 
            startIndex = input.IndexOf('|', endIndex + 1);
        }
 
        foreach (char symbol in inputStringAsCharArray)
        {
            Console.Write(symbol);
        }
 
        Console.WriteLine();
    }
}
